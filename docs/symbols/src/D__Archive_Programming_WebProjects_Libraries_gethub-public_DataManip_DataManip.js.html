 36 /**
 37  * Pivots records on the given field (member). The structure of the result
 38  * is controlled by value of the 'flat' argument.
 39  *
 40  * @example Example
 41  * <code>var data = [
 42  *   { id: 1, name: 'Bob', score: 92 },
 43  *   { id: 2, name: 'Sam', score: 68 },
 44  *   { id: 3, name: 'Bob', score: 89 },
 45  *   { id: 4, name: 'Sam', score: 72 },
 46  *   { id: 5, name: 'Sam', score: 81 }
 47  * ];</code>
 48  *  
 49  * <code>DataManip.pivot( data, 'name' );</code>
 50  * >>>
 51  * <code>{
 52  *   Bob: [
 53  *     { id: 1, name: 'Bob', score: 92 },
 54  *     { id: 3, name: 'Bob', score: 89 }
 55  *   ],
 56  *   Sam: [
 57  *     { id: 2, name: 'Sam', score: 68 },
 58  *     { id: 4, name: 'Sam', score: 72 },
 59  *     { id: 5, name: 'Sam', score: 81 }
 60  *   ]
 61  * ]</code>
 62  * @public
 63  * @memberOf DataManip
 64  * @param {Array} records records to act on
 65  * @param {String} field member name to pivot on
 66  * @param {boolean} [flat=false] controls result structure
 67  * <dl>
 68  *  <dt>true <dd>map of values is returned (value selected is the 'last'
 69  *               one matching)
 70  *  <dt>false<dd>(default) map of arrays of values is returned.
 71  * </dl>
 72  * @returns {Hash} pivoted view of records
 73  */
 74 function pivot( records, field, flat ) {
 75   var out = {};
 76   flat = typeof(flat) == 'undefined' ? false : flat;
 77   
 78   if ( !flat ) {
 79     for( var i = 0, len = records.length; i < len; i ++ ) {
 80       var r = records[i];
 81       // skip bad entries - these occur in IE under the additional case 
 82       // [{},{},] note the trailing ,
 83       if ( !r ) continue; 
 84       var a = out[r[field]];
 85       if ( !a ) {
 86         a = [];
 87         out[r[field]] = a;
 88       }
 89       a.push( r );
 90     }
 91   } else {
 92     for( var i = 0, len = records.length; i < len; i ++ ) {
 93       var r = records[i];
 94       if ( !r ) continue; 
 95       var a = out[r[field]] = r;
 96     }
 97   }
 98   
 99   return out;
100 }

102 /**
103  * Filters records based on the given field name and value.
104  * A record must both have a field with the given name and that fields value
105  * must match the given value to be selected.The data structure does not need
106  * to be consistent (records can have missing or extra fields).
107  *
108  * @example Example
109  * <code>var data = [
110  *   { id: 1, name: 'Bob', score: 92 },
111  *   { id: 2, name: 'Sam', score: 68 },
112  *   { id: 3, name: 'Bob', score: 89 },
113  *   { id: 4, name: 'Sam', score: 72 },
114  *   { id: 5, name: 'Sam', score: 81 }
115  * ];</code>
116  * 
117  * <code>DataManip.filter( data, 'name', 'Bob' );</code>
118  * >>>
119  * <code>[
120  *   { id: 1, name: 'Bob', score: 92 },
121  *   { id: 3, name: 'Bob', score: 89 }
122  * ];</code>
123  *
124  * @public
125  * @memberOf DataManip
126  * @param {Array} records records to act on
127  * @param {String} name of the 'seeking' field
128  * @param value seeking field value
129  * @returns {Array} filtered set of items
130  */
131 function filter( records, name, value ) {
132   var items = [];
133   for( var i = 0, len = records.length; i < len; i ++ ) {
134     var r = records[i];
135     if ( !r ) continue; 
136     var a = r[name];
137     if ( a && a == value ) items.push( r );
138   }
139   return items;
140 }

142 /**
143  * Copy all of the properties in the source objects over to the destination
144  * object. It's in-order, to the last source will override properties of the
145  * same name in previous arguments.
146  * <p>Modified and taken from underscore.js</p>
147  * 
148  * @public
149  * @memberOf DataManip
150  * @param obj object to extend
151  * @param [params...] objects to extend with
152  * @returns obj given (modified)
153  */
154 var extend = function( obj ) {
155   for( var i = 1, len = arguments.length; i < len; i ++ ) {
156     var source = arguments[1];
157     for (var prop in source) obj[prop] = source[prop];
158   }
159   return obj;
160 };

162 /**
163  * Builder of Join functions.
164  * <br />
165  * <dl><lh>Supported joins</lh>
166  *   <dt>FROM
167  *     <dd>(required) used as the base iterator target, every record is visited.
168  *         If occurring more than once treated as CROSS.
169  *   <dt>LEFT JOIN
170  *     <dd>Return all rows from the left table, even if there are no matches in
171  *         the right table.
172  *   <dt>INNER JOIN / JOIN
173  *     <dd>Return rows when there is at least one match in both tables.
174  *   <dt>CROSS JOIN
175  *     <dd>Cartesian product of the sets of rows from the joined tables. Each
176  *         row from the first table is combined with each row from the second
177  *         table and so on.
178  * </dl>
179  *
180  * <ul>
181  * <lh>Notice</lh>
182  * <li>Directives must be listed in dependency order (if c->b->a then directive
183  *     order must be a, b, c). This is true when setting directives using 
184  *     either method (calling .parse or setting .directive).
185  * <li>Directive arguments can be in any order and be non-contiguous
186  * <li>The generated function is completely independent from the builder (once 
187  *     compiled changing the selector will not change the selector used by the
188  *     compiled function but the new selector will be used by any function as
189  *     the product of calling compile again).
190  * <li>The directive value 'join' can be one of: from, left, or inner. one, and
191  *     only one, 'from' is a required directive
192  * <li>Behavior of having multiple from statements is equivalent to performing 
193  *     CROSS JOIN. 
194  * <li>Mixing CROSS joins with other join types may result in unexpected, but 
195  *     deterministic, behavior if/when directive ordering is implemented in the
196  *     future.
197  * </ul>
198  *
199  * @class 
200  * Used to construct and compile SQL like FROM / JOIN statements into a 
201  * javascript function which can then be used for multiple evaluations of similar
202  * data sets.
203  *
204  * @example 
205  * <b>Usage example</b>
206  *   Create instance: 
207  *     <code>var jb = new Join();</code>
208  *
209  *   Set directives in one of the following ways:
210  *     <code>jb.directives = [
211  *       { join: 'from' , arg: 0, as: 'score' },
212  *       { join: 'left' , arg: 1, as: 'grade'  , on: { 'grade': 'id'  , 'score': 'grade_id' } },
213  *       { join: 'inner', arg: 2, as: 'student', on: { 'student': 'id', 'score': 'student_id' } },
214  *       { join: 'left' , arg: 3, as: 'test'   , on: { 'test': 'id'   , 'score': 'test_id' } } 
215  *     ];</code>
216  *   -- OR --
217  *     <code>jb.parse(
218  *       "FROM arg0 AS score " +
219  *       "LEFT JOIN arg1 AS grade   ON grade.id   = score.grade_id " +
220  *       "LEFT JOIN arg2 AS student ON student.id = score.student_id " +
221  *       "LEFT JOIN arg3 AS test    ON test.id    = score.test_id "
222  *     );</code>
223  *
224  *   Set the selector function:
225  *     this selector will result in a set of records containing only test_score less than 70 
226  *     <code>jb.selector = function(){ 
227  *       if ( this.score.test_score >= 70 ) return null;
228  *       return { student_name: this.student.name, grade: this.grade.letter, test_name: this.test.name };
229  *     }</code>
230  *
231  *   Call .compile and either retain the function returned or reference via .exec later
232  *
233  * <b>Calling the generated function</b>
234  *   given lists of simple objects of the same name as aliased in the directive
235  *   example call
236  *     <code>var records = jb.exec( scores, grades, students, tests )</code>
237  *
238  * 
239  * @property {Array} directives 
240  * Directives are used to generate the javascript which will perform the join.
241  * <p>
242  * The directives content is considered 'trusted' and therefore directives are
243  * a possible JS injection vector (sanitize user input before directly
244  * creating directives).
245  * </p><p>
246  * The parse method may be used to create directives for you by using an SQL
247  * fragment like syntax. See: {@link Join#parse}
248  * </p><p>
249  * 
250  * <dl>
251  * <lh>Directives have the following structure</lh>
252  * <dt>join<dd>one of 'from', 'left', 'inner', 'cross'
253  * <dt>arg<dd>index of the function parameter which will be referenced by the
254  *     alias established by 'as'. Does not need to be unique.
255  * <dt>as<dd>Alias which will be used to refer to 'this' view of the specified
256  *     function argument.
257  * <dt>on<dd>Structure containing exactly two members of the form 
258  *     <alias>: <field>, one alias MUST be the same value as 'as'.
259  * </dl>
260  * <b>Example</b>
261  * <pre class='code'>[
262  *   { join: 'from', arg: 0, as: 'score' },
263  *   { join: 'left', arg: 1, as: 'grade'  , on: { 'grade': 'id'  , 'score': 'grade_id' } },
264  *   { join: 'left', arg: 2, as: 'student', on: { 'student': 'id', 'score': 'student_id' } },
265  *   { join: 'left', arg: 3, as: 'test'   , on: { 'test': 'id'   , 'score': 'test_id' } } 
266  * ]</pre>
267  * </p>
268  * <ul>
269  * <lh>Notice</lh>
270  * <li>arg#'s may be reused just as you can join a table on itself in SQL
271  * <li>aliases must be unique within a set of directives
272  * <li>only equality '=' is supported at this time
273  * <li>having a FROM statement is required
274  * <li>joins must have an ON statement that contains reference to the name
275  *     aliased by 'as'
276  * <li>The parse method will throw Error's for common failure cases but this
277  *     error checking may not catch all invalid cases.
278  * <li>Currently directives are not reordered by this system and the order 
279  *     of directives is used to build the join function code. This means that, 
280  *     for now, join dependency MUST be satisfied by the user by the order 
281  *     of listing.
282  * </ul>
283  * 
284  * @property {String} __code last generated JavaScript code
285  * @memberOf DataManip
286  */
287 function Join() {
288   this.__code = undefined;
289   this.directives = [];
290   return;
291 }

315 /**
316  * <b>The user should overwrite this function</b> the default selector simply 
317  * returns the full set of records ('this').
318  * <p>
319  * The selector function like a combination of an SQL SELECT clause and WHERE 
320  * clause. It is used by the compiled join function to perform record selection
321  * as well as creating a record 'view' or transformation.
322  * </p><p>
323  * The selector function is called in the context of the current, joined, row.
324  * In other words this function references 'tables' by their alias through the
325  * 'this' object. For example <code>this.score</code> will refer to the record
326  * that was declared to have the alias of 'score' (via 'as') in the directives.
327  * </p>
328  * <b>Performing 'WHERE' like actions</b>
329  * <p>
330  *   The selector function is capable of handling 'where' clauses. By 
331  *   returning null or undefined the record will be omitted from the result.
332  * </p><p>
333  * Take note that the return value from the selector is pushed into an array
334  * and that array is what is returned from the compiled join function upon
335  * its invocation.
336  * <ul>
337  *   <lh>This has several powerful implications such as...</lh>
338  *   <li>Returning 'this' causes the result record to have the same structure as
339  *       'this' ('this' is cloned and pushed to the result set).
340  *   <li>Returning a contrived anonymous object results in a set of records
341  *       having whatever structure that object defines (e.g. <code>return {name: 
342  *       this.user.name, group: this.group.name};</code>)
343  *   <li>Returning a string or other simple value causes the result record set 
344  *       to simply be a list of these primitives.
345  *   <li>Returning null or undefined results in no value being pushed to the
346  *       result set for this data combination.
347  *   <li>Complex actions such as DOM manipulation can occur within the select
348  *       function in addition to, or instead of, actually building a list of
349  *       records.
350  * </ul>
351  * </p><p>
352  * The selector may be changed after the function is compiled by setting the
353  * bound selector at foo.external.selector where foo is a compiled join 
354  * function. Changing the selector on a Join object will not affect any
355  * previously compiled join functions.
356  * </p>
357  *
358  * @public
359  * @memberOf DataManip.Join.prototype
360  * @returns {Object} Representation of set to capture or null to ignore
361  * this join set.
362  */
363 Join.prototype.selector = function() { return this; }
364 
365 /**
366  * Generates the code to perform the join specified by the directives and
367  * returns a compiled function. Also sets this same function as the member
368  * this.exec
369  * <p>
370  * The compiled function should be called passing a set of arrays matching 
371  * the order specified by the 'arg: #' value of each directive.
372  * </p><p>
373  * Changing the members of exec.external is allowed and makes it possible to
374  * change the select function used by the compiled join function.
375  * </p>
376  * @example
377  *    given directives = [
378  *       { join: 'from', arg: 0, as: 'score' },
379  *       { join: 'left', arg: 1, as: 'grade'  , on: { 'grade': 'id'  , 'score': 'grade_id' } },
380  *       { join: 'left', arg: 2, as: 'student', on: { 'student': 'id', 'score': 'student_id' } },
381  *       { join: 'left', arg: 3, as: 'test'   , on: { 'test': 'id'   , 'score': 'test_id' } } 
382  *     ];
383  *
384  *   given lists of simple objects of the same name as aliased in the directive example
385  *   var records = jb.exec( scores, grades, students, tests );
386  *
387  * @public
388  * @memberOf DataManip.Join.prototype
389  * @returns {Function} function, that when invoked, will perform the join 
390  *          operation specified by the directives as declared.
391  */
392 Join.prototype.compile = function() {
393   this.__code = Join__generate.call( this );
394   // functions/objects  placed within 'external' are made available 
395   // runtime (swappable) via exec.external.* this allows for the selector
396   // function to be changed after this call to compile
397   var context = {
398     external: { 
399       selector: this.selector
400     },
401     extend: extend,
402     pivot: pivot
403   }
404   
405   this.exec = (function( ctx, func ) {
406     var foo = function() {
407       return func.apply( ctx, arguments );
408     }
409     foo.external = ctx.external;
410     return foo;
411   })( context, new Function( this.__code ) );
412   
413   return this.exec;
414 }

416 /**
417  * Generates the javascript code to perform arbitrary join operations 
418  * 
419  * @private
420  * @memberOf DataManip.Join.prototype
421  * @returns {String} JavaScript code
422  */
423 var Join__generate = function() {
424   var argMap = {};
425   var lookupMap = {};
426   var code = '';
427   var eol = this.__eol__;
428   
429   // emit function aliasing
430   code += 'var selector=this.external.selector,' + eol;
431   code +=     'extend=this.extend,' + eol;
432   code +=     'pivot=this.pivot;' + eol;
433   
434   // emit other members
435   code += 'var r=[],' + eol; //records
436   code +=     'x={};' + eol; //context
437   
438   // build list of arguments and lookups needed
439   for( var i = 0, len = this.directives.length; i < len; i ++ ) {
440     var directive = this.directives[i];
441     if ( !directive  ) continue; 
442     var argName = 'arg' + directive.arg;
443     argMap[argName] = argName + '=arguments[' + directive.arg + ']||[]';
444     if ( directive.join != 'from' && directive.join != 'cross') {
445       var indexField = directive.on[ directive.as ];
446       var lookupName = 'p_' + argName + '_' + indexField;
447       lookupMap[lookupName] = lookupName + '=' 
448         + 'pivot(' + argName + ',"' + indexField + '")';
449     }
450   }
451   // emit argument uptake and lookup generation
452   var j = 0;
453   for( var i in argMap ) {
454     if ( j++ == 0 ) code += 'var ';
455     else code += eol + ',';
456     code += argMap[i];
457   }
458   if ( j > 0 ) code += ';' + eol;
459   
460   j = 0;
461   for( var i in lookupMap ) {
462     if ( j++ == 0 ) code += 'var ';
463     else code += eol + ',';
464     code += lookupMap[i];
465   }
466   if ( j > 0 ) code += ';' + eol;
467 
468   // footers are used to 'close' the for loops
469   var footers = [];
470   
471   // generate loop structure to perform join operation
472   for( var i = 0, len = this.directives.length; i < len; i ++ ) {
473     var directive = this.directives[i];
474     if ( !directive  ) continue; 
475     var valuelist_var = 'v' + i;
476     var itr_var = 'i' + i;
477     var itr_len_var = 'L' + i;
478     
479     if ( directive.join == 'left' ) {
480       code += 'var ' + valuelist_var + '=' 
481            + Join__getLookupCommand( directive ) + '||[{}];' + eol;
482     } else if ( directive.join == 'inner' ) {
483       code += 'var ' + valuelist_var + '=' 
484            + Join__getLookupCommand( directive ) + '||[];' + eol;
485     } else if ( directive.join == 'from' || directive.join == 'cross' ) {
486       var argName = 'arg' + directive.arg;
487       code += 'var ' + valuelist_var + '=' + argName + ';' + eol;
488     }
489     code += 'for(var ' + itr_var + '=0,' + itr_len_var + '='
490          + valuelist_var + '.length;'
491          + itr_var + '<' + itr_len_var + ';'
492          + itr_var + '++){' + eol;
493     code += 'x["' + directive.as + '"]=' 
494          + valuelist_var + '[' + itr_var + '];' + eol;
495 
496     // TODO: delete can be removed once directive dependency ordering is ensured
497     footers.push( 'delete(x["' + directive.as + '"]);' + eol + '}' + eol );
498   }
499   
500   // in the deepest loop, emit selector execution
501   code += 'var sr=selector.call(x);' + eol;
502   code += 'if(sr==x)sr=extend({},x);' + eol;
503   code += 'if(sr)r.push(sr);' + eol;
504   
505   // emit loop closing 'footers'
506   while( footers.length > 0 ) {
507     code += footers.pop();
508   }
509   
510   // emit return statement - duh
511   code += 'return r;';
512   return code;
513 }

515 /**
516  * Computes the right hand side of an index based value retrieval
517  * for example the RHS of this statement: var joinWithMeList = lookup[..];
518  * 
519  * @private
520  * @memberOf DataManip.Join
521  * @param directive 
522  * @returns {String} generated lookup access command
523  */
524 function Join__getLookupCommand( directive ) {
525   if ( !directive.on ) return null;
526   var argName = 'arg' + directive.arg;
527   var indexField = directive.on[ directive.as ];
528   var lookupName = 'p_' + argName + '_' + indexField;
529   for ( var i in directive.on ) {
530     if ( i != directive.as ) 
531       return lookupName + '[x["' + i + '"]["' + directive.on[i] + '"]]';
532   }
533 }

536 /**
537  * Parses an SQL like FROM .. JOIN query fragment
538  *
539  * <dl>
540  *   <lh>Groups</lh>
541  *   <td>1<dd>function (from, left join, inner join, cross join, join)
542  *   <td>2<dd>argument number (numeric part of: arg0, arg1, ... 0, 1, ..,)
543  *   <td>3<dd>alias (RHS of AS)
544  *   <td>4<dd>join on LHS alias
545  *   <td>5<dd>join on LHS field
546  *   <td>6<dd>comparison operation (currently can only be '=')
547  *   <td>7<dd>join on RHS alias
548  *   <td>8<dd>join on RHS field
549  * </dl>
550  * @private
551  */
552 var sqlFromJoinRegex = /(FROM|LEFT\s+JOIN|INNER\s+JOIN|CROSS JOIN|JOIN)\s+arg(\d+)\s+AS\s+([$a-zA-Z_][$a-zA-Z0-9_]*)\b(?:\s+ON\s+([$a-zA-Z_][$a-zA-Z0-9_]*)\.([$a-zA-Z_][$a-zA-Z0-9_]*)\s*(=)\s*([$a-zA-Z_][$a-zA-Z0-9_]*)\.([$a-zA-Z_][$a-zA-Z0-9_]*))?/ig;
553 
554 /**
555  * The parse() method allows you to specify directives in a, human friendly,
556  * SQL like syntax consisting of the FROM and JOIN portions of an SQL
557  * statement.
558  * <p>
559  * The 'SELECT' and 'WHERE' parts are handled by the 'selector' user function.
560  * </p>
561  * @notice
562  *    @see Join#directives
563  *  
564  * @example statement which may be passed to parse():
565  *   FROM arg0 AS score
566  *   LEFT JOIN arg1 AS grade   ON grade.id   = score.grade_id
567  *   LEFT JOIN arg2 AS student ON student.id = score.student_id
568  *   LEFT JOIN arg3 AS test    ON test.id    = score.test_id
569  *  
570  * @public
571  * @memberOf DataManip.Join.prototype
572  * @param {String} joinStr SQL join style string to parse
573  * @returns {Join} self for chaining
574  */
575 Join.prototype.parse = function( joinStr ) {
576   this.directives = [];
577   var match;
578   var haveFromStatment = false;
579   var aliasReuseCheck = {};
580   
581   while ( match = sqlFromJoinRegex.exec( joinStr ) ) {
582     var join = match[1].toLowerCase();
583     if ( /^(join|inner\s+join)$/.test( join ) ) join = 'inner';
584     if ( /^left\s+join$/.test( join ) ) join = 'left';
585     if ( /^cross\s+join$/.test( join ) ) join = 'cross';
586     if ( join == 'from' ) {
587       if ( haveFromStatment ) join = 'cross';
588       haveFromStatment = true;
589     }
590     var directive = {
591       join: join,
592       arg: parseInt( match[2] ),
593       as: match[3]
594     };
595     
596     if ( aliasReuseCheck[directive.as] ) {
597       throw new Error( "directive " + this.directives.length + " - alias '" 
598       + directive.as + "' cannot be used more than once" );
599     }
600     aliasReuseCheck[directive.as] = true;
601     
602     if ( !/from|cross/.test( join ) ) {
603       if ( !match[4] || !match[5] || !match[7] || !match[8] )
604         throw new Error( "directive " + this.directives.length 
605         + " - ON statment is missing or incorrectly formatted" );
606       if ( match[4] != directive.as && match[7] != directive.as )
607         throw new Error( "directive " + this.directives.length 
608         + " - ON statment must include a field from " + directive.as );
609       if ( match[4] == match[7] )
610         throw new Error( "directive " + this.directives.length 
611         + " - ON statment must include a field not from " + directive.as );
612       
613       directive.on = {};
614       directive.on[ match[4] ] = match[5];
615       directive.on[ match[7] ] = match[8];
616     }
617     this.directives.push( directive );
618   }
619   
620   if ( !haveFromStatment ) {
621     throw new Error( "FROM statement is required" );
622   }
623   return this;
624 }
